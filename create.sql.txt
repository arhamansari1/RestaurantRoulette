CREATE DATABASE `test_db` /*!40100 DEFAULT CHARACTER SET latin1 */;

CREATE TABLE `test_db`.`users` (
  `user_id` INT NOT NULL AUTO_INCREMENT,
  `first_name` VARCHAR(45) NOT NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `user_email` VARCHAR(99) NOT NULL,
  `user_phone` VARCHAR(15) NOT NULL,
  `username` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`user_id`),
  UNIQUE INDEX `user_id_UNIQUE` (`user_id` ASC) VISIBLE);

CREATE TABLE `test_db`.`res_choices` (
  `room_id` INT NOT NULL,
  `choice1` VARCHAR(45) NOT NULL,
  `choice2` VARCHAR(45) NOT NULL,
  `choice3` VARCHAR(99) NOT NULL
);
